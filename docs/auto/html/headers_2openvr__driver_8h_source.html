1525  * relative to the time of the PoseUpdated() call made by the driver.
1526  */
1527  double poseTimeOffset;
1528 
1529  /* Generally, the pose maintained by a driver
1530  * is in an inertial coordinate system different
1531  * from the world system of x+ right, y+ up, z+ back.
1532  * Also, the driver is not usually tracking the "head" position,
1533  * but instead an internal IMU or another reference point in the HMD.
1534  * The following two transforms transform positions and orientations
1535  * to app world space from driver world space,
1536  * and to HMD head space from driver local body space.
1537  *
1538  * We maintain the driver pose state in its internal coordinate system,
1539  * so we can do the pose prediction math without having to
1540  * use angular acceleration. A driver's angular acceleration is generally not measured,
1541  * and is instead calculated from successive samples of angular velocity.
1542  * This leads to a noisy angular acceleration values, which are also
1543  * lagged due to the filtering required to reduce noise to an acceptable level.
1544  */
1545  vr::HmdQuaternion_t qWorldFromDriverRotation;
1546  double vecWorldFromDriverTranslation[3];
1547 
1548  vr::HmdQuaternion_t qDriverFromHeadRotation;
1549  double vecDriverFromHeadTranslation[3];
1550 
1551  /* State of driver pose, in meters and radians. */
1552  /* Position of the driver tracking reference in driver world space
1553  * +[0] (x) is right
1554  * +[1] (y) is up
1555  * -[2] (z) is forward
1556  */
1557  double vecPosition[3];
1558 
1559  /* Velocity of the pose in meters/second */
1560  double vecVelocity[3];
1561 
1562  /* Acceleration of the pose in meters/second */
1563  double vecAcceleration[3];

1569  * representation. The direction is the angle of
1570  * rotation and the magnitude is the angle around
1571  * that axis in radians/second. */
1572  double vecAngularVelocity[3];
1573 
1574  /* Angular acceleration of the pose in axis-angle
1575  * representation. The direction is the angle of
1576  * rotation and the magnitude is the angle around
1577  * that axis in radians/second^2. */
