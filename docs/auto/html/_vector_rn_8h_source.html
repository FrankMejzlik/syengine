135 // Resize.
136 // If the array is shortened, the information about the allocated length is lost.
137 inline void VectorRn::SetLength(long newLength)
138 {
139  assert(newLength > 0);
140  if (newLength > AllocLength)
141  {
142  delete[] x;
143  AllocLength = Max(newLength, AllocLength << 1);
144  x = new double[AllocLength];
145  }
146  length = newLength;
147 }
