173  btVector3 getRayTo(int x, int y)
174  {
175  CommonRenderInterface* renderer = m_guiHelper->getRenderInterface();
176 
177  if (!renderer)
178  {
179  btAssert(0);
180  return btVector3(0, 0, 0);
181  }
182 
183  float top = 1.f;
184  float bottom = -1.f;
185  float nearPlane = 1.f;
186  float tanFov = (top - bottom) * 0.5f / nearPlane;
187  float fov = btScalar(2.0) * btAtan(tanFov);
188 
189  btVector3 camPos, camTarget;
190 
191  renderer->getActiveCamera()->getCameraPosition(camPos);
192  renderer->getActiveCamera()->getCameraTargetPosition(camTarget);
193 
194  btVector3 rayFrom = camPos;
195  btVector3 rayForward = (camTarget - camPos);
196  rayForward.normalize();
197  float farPlane = 10000.f;
198  rayForward *= farPlane;
199 
200  btVector3 rightOffset;
201  btVector3 cameraUp = btVector3(0, 0, 0);
202  cameraUp[m_guiHelper->getAppInterface()->getUpAxis()] = 1;
203 
204  btVector3 vertical = cameraUp;
205 
206  btVector3 hor;
207  hor = rayForward.cross(vertical);
208  hor.safeNormalize();
209  vertical = hor.cross(rayForward);
210  vertical.safeNormalize();
211 
212  float tanfov = tanf(0.5f * fov);
213 
214  hor *= 2.f * farPlane * tanfov;
215  vertical *= 2.f * farPlane * tanfov;
216 
217  btScalar aspect;
218  float width = float(renderer->getScreenWidth());
219  float height = float(renderer->getScreenHeight());
220 
221  aspect = width / height;
222 
223  hor *= aspect;
224 
225  btVector3 rayToCenter = rayFrom + rayForward;
226  btVector3 dHor = hor * 1.f / width;
227  btVector3 dVert = vertical * 1.f / height;
228 
229  btVector3 rayTo = rayToCenter - 0.5f * hor + 0.5f * vertical;
230  rayTo += btScalar(x) * dHor;
231  rayTo -= btScalar(y) * dVert;
232  return rayTo;
233  }
