149 // fast math routines from Doom3 SDK
150 inline float invSqrt(float x)
151 {
152  float xhalf = 0.5f * x;
153  int i = *(int*)&x; // get bits for floating value
154  i = 0x5f3759df - (i >> 1); // gives initial guess
155  x = *(float*)&i; // convert bits back to float
156  x = x * (1.5f - xhalf * x * x); // Newton step
157  return x;
158 }
