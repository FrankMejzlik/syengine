156 /*
157  * helper macros to generate code that is common to all extensions; caller
158  * should prefix it with static if extension source is in one file; this
159  * could be a utility function, but have to stack 6 unused arguments for
160  * something that is called many, many times would be bad.
161  */
162 #define XEXT_GENERATE_FIND_DISPLAY(proc, extinfo, extname, hooks, nev, data) \
163  XExtDisplayInfo* proc(Display* dpy) \
164  { \
165  XExtDisplayInfo* dpyinfo; \
166  if (!extinfo) \
167  { \
168  if (!(extinfo = XextCreateExtension())) return NULL; \
169  } \
170  if (!(dpyinfo = XextFindDisplay(extinfo, dpy))) \
171  dpyinfo = XextAddDisplay(extinfo, dpy, extname, hooks, nev, data); \
172  return dpyinfo; \
173  }
