142 // Some important concepts to understand to use this library:
143 //
144 // Codepoint
145 // Characters are defined by unicode codepoints, e.g. 65 is
146 // uppercase A, 231 is lowercase c with a cedilla, 0x7e30 is
147 // the hiragana for "ma".
148 //
149 // Glyph
150 // A visual character shape (every codepoint is rendered as
151 // some glyph)
152 //
153 // Glyph index
154 // A font-specific integer ID representing a glyph
155 //
156 // Baseline
157 // Glyph shapes are defined relative to a baseline, which is the
158 // bottom of uppercase characters. Characters extend both above
159 // and below the baseline.
160 //
161 // Current Point
162 // As you draw text to the screen, you keep track of a "current point"
163 // which is the origin of each character. The current point's vertical
164 // position is the baseline. Even "baked fonts" use this model.
165 //
166 // Vertical Font Metrics
167 // The vertical qualities of the font, used to vertically position
168 // and space the characters. See docs for stbtt_GetFontVMetrics.
169 //
170 // Font Size in Pixels or Points
171 // The preferred interface for specifying font sizes in stb_truetype
172 // is to specify how tall the font's vertical extent should be in pixels.
173 // If that sounds good enough, skip the next paragraph.
174 //
175 // Most font APIs instead use "points", which are a common typographic
176 // measurement for describing font size, defined as 72 points per inch.
177 // stb_truetype provides a point API for compatibility. However, true
178 // "per inch" conventions don't make much sense on computer displays
179 // since they different monitors have different number of pixels per
180 // inch. For example, Windows traditionally uses a convention that
181 // there are 96 pixels per inch, thus making 'inch' measurements have
182 // nothing to do with inches, and thus effectively defining a point to
183 // be 1.333 pixels. Additionally, the TrueType font data provides
184 // an explicit scale factor to scale a given font's glyphs to points,
185 // but the author has observed that this scale factor is often wrong
186 // for non-commercial fonts, thus making fonts scaled in points
187 // according to the TrueType spec incoherently sized in practice.
188 //
189 // ADVANCED USAGE
190 //
191 // Quality:
192 //
193 // - Use the functions with Subpixel at the end to allow your characters
194 // to have subpixel positioning. Since the font is anti-aliased, not
195 // hinted, this is very import for quality. (This is not possible with
196 // baked fonts.)
197 //
198 // - Kerning is now supported, and if you're supporting subpixel rendering
199 // then kerning is worth using to give your text a polished look.
200 //
201 // Performance:
202 //
203 // - Convert Unicode codepoints to glyph indexes and operate on the glyphs;
204 // if you don't do this, stb_truetype is forced to do the conversion on
205 // every call.
206 //
207 // - There are a lot of memory allocations. We should modify it to take
208 // a temp buffer and allocate from the temp buffer (without freeing),
209 // should help performance a lot.
210 //
211 // NOTES
212 //
213 // The system uses the raw data found in the .ttf file without changing it
214 // and without building auxiliary data structures. This is a bit inefficient
215 // on little-endian systems (the data is big-endian), but assuming you're
216 // caching the bitmaps or glyph shapes this shouldn't be a big deal.
217 //
218 // It appears to be very hard to programmatically determine what font a
219 // given file is in a general way. I provide an API for this, but I don't
220 // recommend it.

1079 STBTT_DEF int stbtt_FindGlyphIndex(const stbtt_fontinfo *info, int unicode_codepoint)
1080 {
1081  stbtt_uint8 *data = info->data;
1082  stbtt_uint32 index_map = info->index_map;
1083 
1084  stbtt_uint16 format = ttUSHORT(data + index_map + 0);
1085  if (format == 0)
1086  { // apple byte encoding
1087  stbtt_int32 bytes = ttUSHORT(data + index_map + 2);
1088  if (unicode_codepoint < bytes - 6)
1089  return ttBYTE(data + index_map + 6 + unicode_codepoint);
1090  return 0;
1091  }
1092  else if (format == 6)
1093  {
1094  stbtt_uint32 first = ttUSHORT(data + index_map + 6);
1095  stbtt_uint32 count = ttUSHORT(data + index_map + 8);
1096  if ((stbtt_uint32)unicode_codepoint >= first && (stbtt_uint32)unicode_codepoint < first + count)
1097  return ttUSHORT(data + index_map + 10 + (unicode_codepoint - first) * 2);
1098  return 0;
1099  }
1100  else if (format == 2)
1101  {
1102  STBTT_assert(0); // @TODO: high-byte mapping for japanese/chinese/korean
1103  return 0;
1104  }
1105  else if (format == 4)
1106  { // standard mapping for windows fonts: binary search collection of ranges
1107  stbtt_uint16 segcount = ttUSHORT(data + index_map + 6) >> 1;
1108  stbtt_uint16 searchRange = ttUSHORT(data + index_map + 8) >> 1;
1109  stbtt_uint16 entrySelector = ttUSHORT(data + index_map + 10);
1110  stbtt_uint16 rangeShift = ttUSHORT(data + index_map + 12) >> 1;
1111 
1112  // do a binary search of the segments
1113  stbtt_uint32 endCount = index_map + 14;
1114  stbtt_uint32 search = endCount;
1115 
1116  if (unicode_codepoint > 0xffff)
1117  return 0;
1118 
1119  // they lie from endCount .. endCount + segCount
1120  // but searchRange is the nearest power of two, so...
1121  if (unicode_codepoint >= ttUSHORT(data + search + rangeShift * 2))
1122  search += rangeShift * 2;
1123 
1124  // now decrement to bias correctly to find smallest
1125  search -= 2;
1126  while (entrySelector)
1127  {
1128  stbtt_uint16 end;
1129  searchRange >>= 1;
1130  end = ttUSHORT(data + search + searchRange * 2);
1131  if (unicode_codepoint > end)
1132  search += searchRange * 2;
1133  --entrySelector;
1134  }
1135  search += 2;
1136 
1137  {
1138  stbtt_uint16 offset, start;
1139  stbtt_uint16 item = (stbtt_uint16)((search - endCount) >> 1);
1140 
1141  STBTT_assert(unicode_codepoint <= ttUSHORT(data + endCount + 2 * item));
1142  start = ttUSHORT(data + index_map + 14 + segcount * 2 + 2 + 2 * item);
1143  if (unicode_codepoint < start)
1144  return 0;
1145 
1146  offset = ttUSHORT(data + index_map + 14 + segcount * 6 + 2 + 2 * item);
1147  if (offset == 0)
1148  return (stbtt_uint16)(unicode_codepoint + ttSHORT(data + index_map + 14 + segcount * 4 + 2 + 2 * item));
1149 
1150  return ttUSHORT(data + offset + (unicode_codepoint - start) * 2 + index_map + 14 + segcount * 6 + 2 + 2 * item);
1151  }
1152  }
1153  else if (format == 12 || format == 13)
1154  {
1155  stbtt_uint32 ngroups = ttULONG(data + index_map + 12);
1156  stbtt_int32 low, high;
1157  low = 0;
1158  high = (stbtt_int32)ngroups;
1159  // Binary search the right group.
1160  while (low < high)
1161  {
1162  stbtt_int32 mid = low + ((high - low) >> 1); // rounds down, so low <= mid < high
1163  stbtt_uint32 start_char = ttULONG(data + index_map + 16 + mid * 12);
1164  stbtt_uint32 end_char = ttULONG(data + index_map + 16 + mid * 12 + 4);
1165  if ((stbtt_uint32)unicode_codepoint < start_char)
1166  high = mid;
1167  else if ((stbtt_uint32)unicode_codepoint > end_char)
1168  low = mid + 1;
1169  else
1170  {
1171  stbtt_uint32 start_glyph = ttULONG(data + index_map + 16 + mid * 12 + 8);
1172  if (format == 12)
1173  return start_glyph + unicode_codepoint - start_char;
1174  else // format == 13
1175  return start_glyph;
1176  }
1177  }
1178  return 0; // not found
1179  }
1180  // @TODO
1181  STBTT_assert(0);
1182  return 0;
1183 }
