382 B3_FORCE_INLINE b3Scalar b3Sqrt(b3Scalar y)
383 {
384 #ifdef USE_APPROXIMATION
385  double x, z, tempf;
386  unsigned long *tfptr = ((unsigned long *)&tempf) + 1;
387 
388  tempf = y;
389  *tfptr = (0xbfcdd90a - *tfptr) >> 1; /* estimate of 1/sqrt(y) */
390  x = tempf;
391  z = y * b3Scalar(0.5);
392  x = (b3Scalar(1.5) * x) - (x * x) * (x * z); /* iteration formula */
393  x = (b3Scalar(1.5) * x) - (x * x) * (x * z);
394  x = (b3Scalar(1.5) * x) - (x * x) * (x * z);
395  x = (b3Scalar(1.5) * x) - (x * x) * (x * z);
396  x = (b3Scalar(1.5) * x) - (x * x) * (x * z);
397  return x * y;
398 #else
399  return sqrtf(y);
400 #endif
401 }

448 B3_FORCE_INLINE b3Scalar b3Atan2Fast(b3Scalar y, b3Scalar x)
449 {
450  b3Scalar coeff_1 = B3_PI / 4.0f;
451  b3Scalar coeff_2 = 3.0f * coeff_1;
452  b3Scalar abs_y = b3Fabs(y);
453  b3Scalar angle;
454  if (x >= 0.0f)
455  {
456  b3Scalar r = (x - abs_y) / (x + abs_y);
457  angle = coeff_1 - coeff_1 * r;
458  }
459  else
460  {
461  b3Scalar r = (x + abs_y) / (abs_y - x);
462  angle = coeff_2 - coeff_1 * r;
463  }
464  return (y < 0.0f) ? -angle : angle;
465 }

512 B3_FORCE_INLINE unsigned b3Select(unsigned condition, unsigned valueIfConditionNonZero, unsigned valueIfConditionZero)
513 {
514  // Set testNz to 0xFFFFFFFF if condition is nonzero, 0x00000000 if condition is zero
515  // Rely on positive value or'ed with its negative having sign bit on
516  // and zero value or'ed with its negative (which is still zero) having sign bit off
517  // Use arithmetic shift right, shifting the sign bit through all 32 bits
518  unsigned testNz = (unsigned)(((int)condition | -(int)condition) >> 31);
519  unsigned testEqz = ~testNz;
520  return ((valueIfConditionNonZero & testNz) | (valueIfConditionZero & testEqz));
521 }
522 B3_FORCE_INLINE int b3Select(unsigned condition, int valueIfConditionNonZero, int valueIfConditionZero)
523 {
524  unsigned testNz = (unsigned)(((int)condition | -(int)condition) >> 31);
525  unsigned testEqz = ~testNz;
526  return static_cast<int>((valueIfConditionNonZero & testNz) | (valueIfConditionZero & testEqz));
527 }

573 {
