270 inline void ComputeAlignedPoint(const T* const points, const unsigned int idx, const Vec3<double>& barycenter,
271  const double (&rot)[3][3], Vec3<double>& pt){};
272 template <>
273 inline void ComputeAlignedPoint<float>(const float* const points, const unsigned int idx, const Vec3<double>& barycenter, const double (&rot)[3][3], Vec3<double>& pt)
274 {
275  double x = points[idx + 0] - barycenter[0];
276  double y = points[idx + 1] - barycenter[1];
277  double z = points[idx + 2] - barycenter[2];
278  pt[0] = rot[0][0] * x + rot[1][0] * y + rot[2][0] * z;
279  pt[1] = rot[0][1] * x + rot[1][1] * y + rot[2][1] * z;
280  pt[2] = rot[0][2] * x + rot[1][2] * y + rot[2][2] * z;
281 }
282 template <>
283 inline void ComputeAlignedPoint<double>(const double* const points, const unsigned int idx, const Vec3<double>& barycenter, const double (&rot)[3][3], Vec3<double>& pt)
284 {
285  double x = points[idx + 0] - barycenter[0];
286  double y = points[idx + 1] - barycenter[1];
287  double z = points[idx + 2] - barycenter[2];
288  pt[0] = rot[0][0] * x + rot[1][0] * y + rot[2][0] * z;
289  pt[1] = rot[0][1] * x + rot[1][1] * y + rot[2][1] * z;
290  pt[2] = rot[0][2] * x + rot[1][2] * y + rot[2][2] * z;
291 }
292 template <class T>
293 void Volume::ComputeBB(const T* const points, const unsigned int stridePoints, const unsigned int nPoints,
294  const Vec3<double>& barycenter, const double (&rot)[3][3])
295 {
296  Vec3<double> pt;
297  ComputeAlignedPoint(points, 0, barycenter, rot, pt);
298  m_maxBB = pt;
299  m_minBB = pt;
300  for (unsigned int v = 1; v < nPoints; ++v)
301  {
302  ComputeAlignedPoint(points, v * stridePoints, barycenter, rot, pt);
303  for (int i = 0; i < 3; ++i)
304  {
305  if (pt[i] < m_minBB[i])
306  m_minBB[i] = pt[i];
307  else if (pt[i] > m_maxBB[i])
308  m_maxBB[i] = pt[i];
309  }
310  }
311 }
312 template <class T>
313 void Volume::Voxelize(const T* const points, const unsigned int stridePoints, const unsigned int nPoints,
314  const int* const triangles, const unsigned int strideTriangles, const unsigned int nTriangles,
315  const size_t dim, const Vec3<double>& barycenter, const double (&rot)[3][3])
316 {
317  if (nPoints == 0)
318  {
319  return;
320  }
321  ComputeBB(points, stridePoints, nPoints, barycenter, rot);
322 
323  double d[3] = {m_maxBB[0] - m_minBB[0], m_maxBB[1] - m_minBB[1], m_maxBB[2] - m_minBB[2]};
324  double r;
325  if (d[0] > d[1] && d[0] > d[2])
326  {
327  r = d[0];
328  m_dim[0] = dim;
329  m_dim[1] = 2 + static_cast<size_t>(dim * d[1] / d[0]);
330  m_dim[2] = 2 + static_cast<size_t>(dim * d[2] / d[0]);
331  }
332  else if (d[1] > d[0] && d[1] > d[2])
333  {
334  r = d[1];
335  m_dim[1] = dim;
336  m_dim[0] = 2 + static_cast<size_t>(dim * d[0] / d[1]);
337  m_dim[2] = 2 + static_cast<size_t>(dim * d[2] / d[1]);
338  }
339  else
340  {
341  r = d[2];
342  m_dim[2] = dim;
343  m_dim[0] = 2 + static_cast<size_t>(dim * d[0] / d[2]);
344  m_dim[1] = 2 + static_cast<size_t>(dim * d[1] / d[2]);
345  }
346 
347  m_scale = r / (dim - 1);
348  double invScale = (dim - 1) / r;
349 
350  Allocate();
351  m_numVoxelsOnSurface = 0;
352  m_numVoxelsInsideSurface = 0;
353  m_numVoxelsOutsideSurface = 0;
354 
355  Vec3<double> p[3];
356  size_t i, j, k;
357  size_t i0, j0, k0;
358  size_t i1, j1, k1;
359  Vec3<double> boxcenter;
360  Vec3<double> pt;
361  const Vec3<double> boxhalfsize(0.5, 0.5, 0.5);
362  for (size_t t = 0, ti = 0; t < nTriangles; ++t, ti += strideTriangles)
363  {
364  Vec3<int> tri(triangles[ti + 0],
365  triangles[ti + 1],
366  triangles[ti + 2]);
367  for (int c = 0; c < 3; ++c)
368  {
369  ComputeAlignedPoint(points, tri[c] * stridePoints, barycenter, rot, pt);
370  p[c][0] = (pt[0] - m_minBB[0]) * invScale;
371  p[c][1] = (pt[1] - m_minBB[1]) * invScale;
372  p[c][2] = (pt[2] - m_minBB[2]) * invScale;
373  i = static_cast<size_t>(p[c][0] + 0.5);
374  j = static_cast<size_t>(p[c][1] + 0.5);
375  k = static_cast<size_t>(p[c][2] + 0.5);
376  assert(i < m_dim[0] && i >= 0 && j < m_dim[1] && j >= 0 && k < m_dim[2] && k >= 0);
377 
378  if (c == 0)
379  {
380  i0 = i1 = i;
381  j0 = j1 = j;
382  k0 = k1 = k;
383  }
384  else
385  {
386  if (i < i0)
387  i0 = i;
388  if (j < j0)
389  j0 = j;
390  if (k < k0)
391  k0 = k;
392  if (i > i1)
393  i1 = i;
394  if (j > j1)
395  j1 = j;
396  if (k > k1)
397  k1 = k;
398  }
399  }
400  if (i0 > 0)
401  --i0;
402  if (j0 > 0)
403  --j0;
404  if (k0 > 0)
405  --k0;
406  if (i1 < m_dim[0])
407  ++i1;
408  if (j1 < m_dim[1])
409  ++j1;
410  if (k1 < m_dim[2])
411  ++k1;
412  for (size_t i = i0; i < i1; ++i)
413  {
414  boxcenter[0] = (double)i;
415  for (size_t j = j0; j < j1; ++j)
416  {
417  boxcenter[1] = (double)j;
418  for (size_t k = k0; k < k1; ++k)
419  {
420  boxcenter[2] = (double)k;
421  int res = TriBoxOverlap(boxcenter, boxhalfsize, p[0], p[1], p[2]);
422  unsigned char& value = GetVoxel(i, j, k);
423  if (res == 1 && value == PRIMITIVE_UNDEFINED)
424  {
425  value = PRIMITIVE_ON_SURFACE;
426  ++m_numVoxelsOnSurface;
427  }
428  }
429  }
430  }
431  }
432  FillOutsideSurface(0, 0, 0, m_dim[0], m_dim[1], 1);
433  FillOutsideSurface(0, 0, m_dim[2] - 1, m_dim[0], m_dim[1], m_dim[2]);
434  FillOutsideSurface(0, 0, 0, m_dim[0], 1, m_dim[2]);
435  FillOutsideSurface(0, m_dim[1] - 1, 0, m_dim[0], m_dim[1], m_dim[2]);
436  FillOutsideSurface(0, 0, 0, 1, m_dim[1], m_dim[2]);
437  FillOutsideSurface(m_dim[0] - 1, 0, 0, m_dim[0], m_dim[1], m_dim[2]);
438  FillInsideSurface();
439 }
