208  btVector3 getRayTo(int x, int y)
209  {
210  CommonRenderInterface* renderer = m_guiHelper->getRenderInterface();
211 
212  if (!renderer)
213  {
214  btAssert(0);
215  return btVector3(0, 0, 0);
216  }
217 
218  float top = 1.f;
219  float bottom = -1.f;
220  float nearPlane = 1.f;
221  float tanFov = (top - bottom) * 0.5f / nearPlane;
222  float fov = btScalar(2.0) * btAtan(tanFov);
223 
224  btVector3 camPos, camTarget;
225 
226  renderer->getActiveCamera()->getCameraPosition(camPos);
227  renderer->getActiveCamera()->getCameraTargetPosition(camTarget);
228 
229  btVector3 rayFrom = camPos;
230  btVector3 rayForward = (camTarget - camPos);
231  rayForward.normalize();
232  float farPlane = 10000.f;
233  rayForward *= farPlane;
234 
235  btVector3 rightOffset;
236  btVector3 cameraUp = btVector3(0, 0, 0);
237  cameraUp[m_guiHelper->getAppInterface()->getUpAxis()] = 1;
238 
239  btVector3 vertical = cameraUp;
240 
241  btVector3 hor;
242  hor = rayForward.cross(vertical);
243  hor.safeNormalize();
244  vertical = hor.cross(rayForward);
245  vertical.safeNormalize();
246 
247  float tanfov = tanf(0.5f * fov);
248 
249  hor *= 2.f * farPlane * tanfov;
250  vertical *= 2.f * farPlane * tanfov;
251 
252  btScalar aspect;
253  float width = float(renderer->getScreenWidth());
254  float height = float(renderer->getScreenHeight());
255 
256  aspect = width / height;
257 
258  hor *= aspect;
259 
260  btVector3 rayToCenter = rayFrom + rayForward;
261  btVector3 dHor = hor * 1.f / width;
262  btVector3 dVert = vertical * 1.f / height;
263 
264  btVector3 rayTo = rayToCenter - 0.5f * hor + 0.5f * vertical;
265  rayTo += btScalar(x) * dHor;
266  rayTo -= btScalar(y) * dVert;
267  return rayTo;
268  }
