176 inline void clampToCustomSpeedNotches(float speed, void*)
177 { // function to clamp to custom speed notches
178  double minSpeed = 0;
179  double minSpeedDist = SimulationSpeeds::MAX_SPEED;
180  for (int i = 0; i < SimulationSpeeds::NUM_SPEEDS; i++)
181  {
182  double speedDist = (speeds[i] - speed >= 0) ? speeds[i] - speed : speed - speeds[i]; // float absolute
183 
184  if (minSpeedDist > speedDist)
185  {
186  minSpeedDist = speedDist;
187  minSpeed = speeds[i];
188  }
189  }
190  gSimulationSpeed = minSpeed;
191 }

453  void createEmptyDynamicsWorld()
454  { // create an empty dynamics worlds according to the chosen settings via statics (top section of code)
455 
457  m_collisionConfiguration = new btDefaultCollisionConfiguration();
458  //m_collisionConfiguration->setConvexConvexMultipointIterations();
459 
461  m_dispatcher = new btCollisionDispatcher(m_collisionConfiguration);
462 
463  // default broadphase
464  m_broadphase = new btDbvtBroadphase();
465 
466  // different solvers require different settings
467  switch (SOLVER_TYPE)
468  {
469  case SEQUENTIALIMPULSESOLVER:
470  {
471  // b3Printf("=%s=",SolverType::SEQUENTIALIMPULSESOLVER);
472  m_solver = new btSequentialImpulseConstraintSolver();
473  break;
474  }
475  case NNCGSOLVER:
476  {
477  // b3Printf("=%s=",SolverType::NNCGSOLVER);
478  m_solver = new btNNCGConstraintSolver();
479  break;
480  }
481  case DANZIGSOLVER:
482  {
483  // b3Printf("=%s=",SolverType::DANZIGSOLVER);
484  btDantzigSolver* mlcp = new btDantzigSolver();
485  m_solver = new btMLCPSolver(mlcp);
486  break;
487  }
488  case GAUSSSEIDELSOLVER:
489  {
490  // b3Printf("=%s=",SolverType::GAUSSSEIDELSOLVER);
491  btSolveProjectedGaussSeidel* mlcp = new btSolveProjectedGaussSeidel();
492  m_solver = new btMLCPSolver(mlcp);
493  break;
494  }
495  case LEMKESOLVER:
496  {
497  // b3Printf("=%s=",SolverType::LEMKESOLVER);
498  btLemkeSolver* mlcp = new btLemkeSolver();
499  m_solver = new btMLCPSolver(mlcp);
500  break;
501  }
502  case FSSOLVER:
503  {
504  // b3Printf("=%s=",SolverType::FSSOLVER);
505  //Use the btMultiBodyConstraintSolver for Featherstone btMultiBody support
506  m_solver = new btMultiBodyConstraintSolver;
507 
508  break;
509  }
510  default:
511  break;
512  }
513 
514  if (SOLVER_TYPE != FSSOLVER)
515  {
516  //TODO: Set parameters for other solvers
517 
518  m_dynamicsWorld = new btDiscreteDynamicsWorld(m_dispatcher,
519  m_broadphase, m_solver, m_collisionConfiguration);
520 
521  if (SOLVER_TYPE == DANZIGSOLVER || SOLVER_TYPE == GAUSSSEIDELSOLVER)
522  {
523  m_dynamicsWorld->getSolverInfo().m_minimumSolverBatchSize = 1; //for mlcp solver it is better to have a small A matrix
524  }
525  else
526  {
527  m_dynamicsWorld->getSolverInfo().m_minimumSolverBatchSize = 128; //for direct solver, it is better to solve multiple objects together, small batches have high overhead
528  }
529 
530  m_dynamicsWorld->getDispatchInfo().m_useContinuous = true; // set continuous collision
531  }
532  else
533  {
534  //use btMultiBodyDynamicsWorld for Featherstone btMultiBody support
535  m_dynamicsWorld = new btMultiBodyDynamicsWorld(m_dispatcher,
536  m_broadphase, (btMultiBodyConstraintSolver*)m_solver,
537  m_collisionConfiguration);
538  }
539 
540  changeERPCFM(); // set appropriate ERP/CFM values according to the string and damper properties of the constraint
541 
542  if (useSplitImpulse)
543  { // If you experience strong repulsion forces in your constraints, it might help to enable the split impulse feature
544  m_dynamicsWorld->getSolverInfo().m_splitImpulse = 1; //enable split impulse feature
545  // m_dynamicsWorld->getSolverInfo().m_splitImpulsePenetrationThreshold =
546  // -0.02;
547  // m_dynamicsWorld->getSolverInfo().m_erp2 = BulletUtils::getERP(
548  // fixedPhysicsStepSizeSec, 10, 1);
549  // m_dynamicsWorld->getSolverInfo().m_splitImpulseTurnErp =
550  // BulletUtils::getERP(fixedPhysicsStepSizeSec, 10, 1);
551  // b3Printf("Using split impulse feature with ERP/TurnERP: (%f,%f)",
552  // m_dynamicsWorld->getSolverInfo().m_erp2,
553  // m_dynamicsWorld->getSolverInfo().m_splitImpulseTurnErp);
554  }
555 
556  m_dynamicsWorld->getSolverInfo().m_numIterations = gSolverIterations; // set the number of solver iterations for iteration based solvers
557 
558  m_dynamicsWorld->setGravity(btVector3(0, -9.81f, 0)); // set gravity to -9.81
559  }

575  void stepSimulation(float deltaTime)
576  { // customly step the simulation
577  do
578  {
579  // // settings
580  if (mPhysicsStepsPerSecondUpdated)
581  {
582  changePhysicsStepsPerSecond(gPhysicsStepsPerSecond);
583  mPhysicsStepsPerSecondUpdated = false;
584  }
585 
586  if (mFramesPerSecondUpdated)
587  {
588  changeFPS(gFramesPerSecond);
589  mFramesPerSecondUpdated = false;
590  }
591 
592  if (gChangeErpCfm)
593  {
594  changeERPCFM();
595  gChangeErpCfm = false;
596  }
597 
598  if (mSolverIterationsUpdated)
599  {
600  changeSolverIterations(gSolverIterations);
601  mSolverIterationsUpdated = false;
602  }
603 
604  // structure according to the canonical game loop
605  // http://www.bulletphysics.org/mediawiki-1.5.8/index.php/Canonical_Game_Loop
606 
607  //##############
608  // breaking conditions - if the loop should stop, then check it here
609 
610  //#############
611  // model update - here you perform updates of your model, be it the physics model, the game or simulation state or anything not related to graphics and input
612 
613  timeWarpSimulation(deltaTime);
614  if (mLoopTimer.getTimeSeconds() - speedUpPrintTimeStamp > 1)
615  {
616  // on reset, we calculate the performed speed up
617  //double speedUp = ((double)performedTime*1000.0)/((double)(mLoopTimer.getTimeMilliseconds()-performanceTimestamp));
618  // b3Printf("Avg Effective speedup: %f",speedUp);

789  void performTrueSteps(btScalar timeStep)
790  { // physics stepping without interpolated substeps
791  int subSteps = floor((timeStep / fixedPhysicsStepSizeSec) + 0.5);
793  for (int i = 0; i < subSteps; i++)
794  {
795  if (timeStep && m_dynamicsWorld)
796  {
797  // since we want to perform all proper steps, we perform no interpolated substeps
798  int subSteps = 1;
799 
800  m_dynamicsWorld->stepSimulation(btScalar(timeStep),
801  btScalar(subSteps), btScalar(fixedPhysicsStepSizeSec));
802  }
803  }
804  }
805 
806  void performInterpolatedSteps(btScalar timeStep)
807  { // physics stepping with interpolated substeps
808  int subSteps = 1 + floor((timeStep / fixedPhysicsStepSizeSec) + 0.5);
809  if (timeStep && m_dynamicsWorld)
810  {
811  m_dynamicsWorld->stepSimulation(btScalar(timeStep), btScalar(subSteps),
812  btScalar(fixedPhysicsStepSizeSec));
813  }
814  }

816  void performMaxStep()
817  { // perform as many steps as possible
818  if (gApplicationTick >= mLastGraphicsTick + mLastInputTick)
819  { // if the remaining time for graphics is going to be positive
820  mPhysicsTick = gApplicationTick
821  - mLastGraphicsTick - mLastInputTick;
822  }
823  else
824  {
825  mPhysicsTick = 0; // no time for physics left / The internal application step is too high
826  }
827 
828  // b3Printf("Application tick: %u",gApplicationTick);
829  // b3Printf("Graphics tick: %u",mLastGraphicsTick);
830  // b3Printf("Input tick: %u",mLastInputTick);
831  // b3Printf("Physics tick: %u",mPhysicsTick);
832 
833  if (mPhysicsTick > 0)
834  { // with positive physics tick we perform as many update steps until the time for it is used up
835 
836  mPhysicsStepStart = mLoopTimer.getTimeMilliseconds();
837  mPhysicsStepEnd = mPhysicsStepStart;
838 
839  while (mPhysicsTick > mPhysicsStepEnd - mPhysicsStepStart)
840  {
841  // b3Printf("Physics passed: %u", mPhysicsStepEnd - mPhysicsStepStart);
842  double timeStep = fixedPhysicsStepSizeSec;
844  if (gInterpolate)
845  {
846  performInterpolatedSteps(timeStep);
847  }
848  else
849  {
850  performTrueSteps(timeStep);
851  }
852  performedTime += timeStep;
853  mPhysicsStepEnd = mLoopTimer.getTimeMilliseconds();
854  }
855  }
856  }

858  void performSpeedStep()
859  { // force-perform the number of steps needed to achieve a certain speed (safe to too high speeds, meaning the application will lose time, not the physics)
860  if (mFrameTime > gApplicationTick)
861  {
862  mFrameTime = gApplicationTick; // This prevents the physics time accumulator to sum up too much time
863  } // The simulation therefore gets slower, but still performs all requested physics steps
864 
865  mModelAccumulator += mFrameTime;
866  // b3Printf("Model time accumulator: %u", mModelAccumulator);
867 
868  int steps = floor(mModelAccumulator / fixedPhysicsStepSizeMilli);
869  // b3Printf("Next steps: %i", steps);
870 
871  if (steps > 0)
872  {
874  double timeStep = gSimulationSpeed * steps * fixedPhysicsStepSizeSec;
876  if (gInterpolate)
877  {
878  performInterpolatedSteps(timeStep); // perform interpolated steps
879  }
880  else
881  {
882  performTrueSteps(timeStep); // perform full steps
883  }
884  performedTime += timeStep; // sum up the performed time for measuring the speed up
885  mModelAccumulator -= steps * fixedPhysicsStepSizeMilli;
886  }
887  }
