209 inline bool LessOrEqualFuzzy(double x, double y)
210 {
211  if (x <= y)
212  {
213  return true;
214  }
215 
216  if (y > 0.0)
217  {
218  if (x > 0.0)
219  {
220  return (x * OneMinusEpsilon15 < y * OnePlusEpsilon15);
221  }
222  else
223  {
224  return (y < 1.0e-15); // x==0 in this case
225  }
226  }
227  else if (y < 0.0)
228  {
229  if (x < 0.0)

231  return (x * OnePlusEpsilon15 < y * OneMinusEpsilon15);
232  }
233  else

236  }
237  }
238  else
239  {
240  return (-1.0e-15 < x && x < 1.0e-15);
241  }
242 }
