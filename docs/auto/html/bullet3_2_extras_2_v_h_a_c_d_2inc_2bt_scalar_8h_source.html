258 SIMD_FORCE_INLINE btScalar btSqrt(btScalar y)
259 {
260 #ifdef USE_APPROXIMATION
261  double x, z, tempf;
262  unsigned long* tfptr = ((unsigned long*)&tempf) + 1;
263 
264  tempf = y;
265  *tfptr = (0xbfcdd90a - *tfptr) >> 1; /* estimate of 1/sqrt(y) */
266  x = tempf;
267  z = y * btScalar(0.5);
268  x = (btScalar(1.5) * x) - (x * x) * (x * z); /* iteration formula */
269  x = (btScalar(1.5) * x) - (x * x) * (x * z);
270  x = (btScalar(1.5) * x) - (x * x) * (x * z);
271  x = (btScalar(1.5) * x) - (x * x) * (x * z);
272  x = (btScalar(1.5) * x) - (x * x) * (x * z);
273  return x * y;
274 #else
275  return sqrtf(y);
276 #endif
277 }

324 SIMD_FORCE_INLINE btScalar btAtan2Fast(btScalar y, btScalar x)
325 {
326  btScalar coeff_1 = SIMD_PI / 4.0f;
327  btScalar coeff_2 = 3.0f * coeff_1;
328  btScalar abs_y = btFabs(y);
329  btScalar angle;
330  if (x >= 0.0f)
331  {
332  btScalar r = (x - abs_y) / (x + abs_y);
333  angle = coeff_1 - coeff_1 * r;
334  }
335  else
336  {
337  btScalar r = (x + abs_y) / (abs_y - x);
338  angle = coeff_2 - coeff_1 * r;
339  }
340  return (y < 0.0f) ? -angle : angle;
341 }

388 SIMD_FORCE_INLINE unsigned btSelect(unsigned condition, unsigned valueIfConditionNonZero, unsigned valueIfConditionZero)
389 {
390  // Set testNz to 0xFFFFFFFF if condition is nonzero, 0x00000000 if condition is zero
391  // Rely on positive value or'ed with its negative having sign bit on
392  // and zero value or'ed with its negative (which is still zero) having sign bit off
393  // Use arithmetic shift right, shifting the sign bit through all 32 bits
394  unsigned testNz = (unsigned)(((int)condition | -(int)condition) >> 31);
395  unsigned testEqz = ~testNz;
396  return ((valueIfConditionNonZero & testNz) | (valueIfConditionZero & testEqz));
397 }

402  return static_cast<int>((valueIfConditionNonZero & testNz) | (valueIfConditionZero & testEqz));
403 }
