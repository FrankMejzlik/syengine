209 /*
210  A dynamic array of Plain Old Data. Doesn't support constructors, etc.
211  Has a small initial memory pool, so that low or no usage will not
212  cause a call to new/delete
213 */
214 template <class T, int INITIAL_SIZE>
215 class DynArray
216 {
217 public:
218  DynArray() : _mem(_pool),
219  _allocated(INITIAL_SIZE),
220  _size(0)
221  {
222  }
223 
224  ~DynArray()
225  {
226  if (_mem != _pool)
227  {
228  delete[] _mem;
229  }
230  }
231 
232  void Clear()
233  {
234  _size = 0;
235  }
236 
237  void Push(T t)
238  {
239  TIXMLASSERT(_size < INT_MAX);
240  EnsureCapacity(_size + 1);
241  _mem[_size] = t;
242  ++_size;
243  }
244 
245  T* PushArr(int count)
246  {
247  TIXMLASSERT(count >= 0);
248  TIXMLASSERT(_size <= INT_MAX - count);
249  EnsureCapacity(_size + count);
250  T* ret = &_mem[_size];
251  _size += count;
252  return ret;
253  }
254 
255  T Pop()
256  {
257  TIXMLASSERT(_size > 0);
258  --_size;
259  return _mem[_size];
260  }
261 
262  void PopArr(int count)
263  {
264  TIXMLASSERT(_size >= count);
265  _size -= count;
266  }
267 
268  bool Empty() const
269  {
270  return _size == 0;
271  }
272 
273  T& operator[](int i)
274  {
275  TIXMLASSERT(i >= 0 && i < _size);
276  return _mem[i];
277  }
278 
279  const T& operator[](int i) const
280  {
281  TIXMLASSERT(i >= 0 && i < _size);
282  return _mem[i];
283  }
284 
285  const T& PeekTop() const
286  {
287  TIXMLASSERT(_size > 0);
288  return _mem[_size - 1];
289  }
290 
291  int Size() const
292  {
293  TIXMLASSERT(_size >= 0);
294  return _size;
295  }
296 
297  int Capacity() const
298  {
299  TIXMLASSERT(_allocated >= INITIAL_SIZE);
300  return _allocated;
301  }
302 
303  void SwapRemove(int i)
304  {
305  TIXMLASSERT(i >= 0 && i < _size);
306  TIXMLASSERT(_size > 0);
307  _mem[i] = _mem[_size - 1];
308  --_size;
309  }
310 
311  const T* Mem() const
312  {
313  TIXMLASSERT(_mem);
314  return _mem;
315  }
316 
317  T* Mem()
318  {
319  TIXMLASSERT(_mem);
320  return _mem;
321  }
322 
323 private:
324  DynArray(const DynArray&); // not supported
325  void operator=(const DynArray&); // not supported
326 
327  void EnsureCapacity(int cap)
328  {
329  TIXMLASSERT(cap > 0);
330  if (cap > _allocated)
331  {
332  TIXMLASSERT(cap <= INT_MAX / 2);
333  int newAllocated = cap * 2;
334  T* newMem = new T[newAllocated];
335  TIXMLASSERT(newAllocated >= _size);
336  memcpy(newMem, _mem, sizeof(T) * _size); // warning: not using constructors, only works for PODs
337  if (_mem != _pool)
338  {
339  delete[] _mem;
340  }
341  _mem = newMem;
342  _allocated = newAllocated;
343  }
344  }
345 
346  T* _mem;
347  T _pool[INITIAL_SIZE];
348  int _allocated; // objects allocated
349  int _size; // number objects in use
350 };

369 /*
370  Template child class to create pools of the correct type.
371 */
372 template <int ITEM_SIZE>
373 class MemPoolT : public MemPool
374 {
375 public:
376  MemPoolT() : _blockPtrs(), _root(0), _currentAllocs(0), _nAllocs(0), _maxAllocs(0), _nUntracked(0) {}
377  ~MemPoolT()
378  {
379  Clear();
380  }
381 
382  void Clear()
383  {
384  // Delete the blocks.
385  while (!_blockPtrs.Empty())
386  {
387  Block* lastBlock = _blockPtrs.Pop();
388  delete lastBlock;
389  }
390  _root = 0;
391  _currentAllocs = 0;
392  _nAllocs = 0;
393  _maxAllocs = 0;
394  _nUntracked = 0;
395  }
396 
397  virtual int ItemSize() const
398  {
399  return ITEM_SIZE;
400  }
401  int CurrentAllocs() const
402  {
403  return _currentAllocs;
404  }
405 
406  virtual void* Alloc()
407  {
408  if (!_root)
409  {
410  // Need a new block.
411  Block* block = new Block();
412  _blockPtrs.Push(block);
413 
414  Item* blockItems = block->items;
415  for (int i = 0; i < ITEMS_PER_BLOCK - 1; ++i)
416  {
417  blockItems[i].next = &(blockItems[i + 1]);
418  }
419  blockItems[ITEMS_PER_BLOCK - 1].next = 0;
420  _root = blockItems;
421  }
422  Item* const result = _root;
423  TIXMLASSERT(result != 0);
424  _root = _root->next;
425 
426  ++_currentAllocs;
427  if (_currentAllocs > _maxAllocs)
428  {
429  _maxAllocs = _currentAllocs;
430  }
431  ++_nAllocs;
432  ++_nUntracked;
433  return result;
434  }
435 
436  virtual void Free(void* mem)
437  {
438  if (!mem)
439  {
440  return;
441  }
442  --_currentAllocs;
443  Item* item = static_cast<Item*>(mem);
444 #ifdef TINYXML2_DEBUG
445  memset(item, 0xfe, sizeof(*item));
446 #endif
447  item->next = _root;
448  _root = item;
449  }
450  void Trace(const char* name)
451  {
452  printf("Mempool %s watermark=%d [%dk] current=%d size=%d nAlloc=%d blocks=%d\n",
453  name, _maxAllocs, _maxAllocs * ITEM_SIZE / 1024, _currentAllocs,
454  ITEM_SIZE, _nAllocs, _blockPtrs.Size());
455  }
456 
457  void SetTracked()
458  {
459  --_nUntracked;
460  }
461 
462  int Untracked() const
463  {
464  return _nUntracked;
465  }
466 
467  // This number is perf sensitive. 4k seems like a good tradeoff on my machine.
468  // The test file is large, 170k.
469  // Release: VS2010 gcc(no opt)
470  // 1k: 4000
471  // 2k: 4000
472  // 4k: 3900 21000
473  // 16k: 5200
474  // 32k: 4300
475  // 64k: 4000 21000
476  // Declared public because some compilers do not accept to use ITEMS_PER_BLOCK
477  // in private part if ITEMS_PER_BLOCK is private
478  enum
479  {
480  ITEMS_PER_BLOCK = (4 * 1024) / ITEM_SIZE
481  };
482 
483 private:
484  MemPoolT(const MemPoolT&); // not supported
485  void operator=(const MemPoolT&); // not supported
486 
487  union Item {
488  Item* next;
489  char itemData[ITEM_SIZE];
490  };
491  struct Block
492  {
493  Item items[ITEMS_PER_BLOCK];
494  };
495  DynArray<Block*, 10> _blockPtrs;
496  Item* _root;
497

499  int _nAllocs;
