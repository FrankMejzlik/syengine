206  float ComputeNormal(const Vector3d &A,
207  const Vector3d &B,
208  const Vector3d &C)
209  {
210  float vx, vy, vz, wx, wy, wz, vw_x, vw_y, vw_z, mag;
211 
212  vx = (B.x - C.x);
213  vy = (B.y - C.y);
214  vz = (B.z - C.z);
215 
216  wx = (A.x - B.x);
217  wy = (A.y - B.y);
218  wz = (A.z - B.z);
219 
220  vw_x = vy * wz - vz * wy;
221  vw_y = vz * wx - vx * wz;
222  vw_z = vx * wy - vy * wx;
223 
224  mag = sqrtf((vw_x * vw_x) + (vw_y * vw_y) + (vw_z * vw_z));
225 
226  if (mag < 0.000001f)
227  {
228  mag = 0;
229  }
230  else
231  {
232  mag = 1.0f / mag;
233  }
234 
235  x = vw_x * mag;
236  y = vw_y * mag;
237  z = vw_z * mag;
238 
239  return mag;
240  }

324  // Highly specialized interpolate routine. Will compute the interpolated position
325  // shifted forward or backwards along the ray defined between (from) and (to).
326  // Reason for existance is so that when a bullet collides with a wall, for
327  // example, you can generate a graphic effect slightly *before* it hit the
328  // wall so that the effect doesn't sort into the wall itself.
329  void Interpolate(const Vector3d &from, const Vector3d &to, float offset)
330  {
331  x = to.x - from.x;
332  y = to.y - from.y;
333  z = to.z - from.z;
334  float d = sqrtf(x * x + y * y + z * z);
335  float recip = 1.0f / d;
336  x *= recip;
337  y *= recip;
338  z *= recip; // normalize vector
339  d += offset; // shift along ray
340  x = x * d + from.x;
341  y = y * d + from.y;
342  z = z * d + from.z;
343  };

564  /******************************************/
565  // Check if next edge (b to c) turns inward
566  //
567  // Edge from a to b is already in face
568  // Edge from b to c is being considered for addition to face
569  /******************************************/
570  bool Concave(const Vector3d &a, const Vector3d &b)
571  {
572  float vx, vy, vz, wx, wy, wz, vw_x, vw_y, vw_z, mag, nx, ny, nz, mag_a, mag_b;
573 
574  wx = b.x - a.x;
575  wy = b.y - a.y;
576  wz = b.z - a.z;
577 
578  mag_a = (float)sqrtf((wx * wx) + (wy * wy) + (wz * wz));
579 
580  vx = x - b.x;
581  vy = y - b.y;
582  vz = z - b.z;
583 
584  mag_b = (float)sqrtf((vx * vx) + (vy * vy) + (vz * vz));
585 
586  vw_x = (vy * wz) - (vz * wy);
587  vw_y = (vz * wx) - (vx * wz);
588  vw_z = (vx * wy) - (vy * wx);
589 
590  mag = (float)sqrtf((vw_x * vw_x) + (vw_y * vw_y) + (vw_z * vw_z));
591 
592  // Check magnitude of cross product, which is a sine function
593  // i.e., mag (a x b) = mag (a) * mag (b) * sin (theta);
594  // If sin (theta) small, then angle between edges is very close to
595  // 180, which we may want to call a concavity. Setting the
596  // CONCAVITY_TOLERANCE value greater than about 0.01 MAY cause
597  // face consolidation to get stuck on particular face. Most meshes
598  // convert properly with a value of 0.0
599 
600  if (mag / (mag_a * mag_b) <= 0.0f) return true;
601 
602  mag = 1.0f / mag;
603 
604  nx = vw_x * mag;
605  ny = vw_y * mag;
606  nz = vw_z * mag;
607 
608  // Dot product of tri normal with cross product result will
609  // yield positive number if edges are convex (+1.0 if two tris
610  // are coplanar), negative number if edges are concave (-1.0 if
611  // two tris are coplanar.)
612 
613  mag = (x * nx) + (y * ny) + (z * nz);
614 
615  if (mag > 0.0f) return false;
616 
617  return (true);
618  };

630  bool PointInTriXY(const Vector3d &p1,
631  const Vector3d &p2,
632  const Vector3d &p3) const
633  {
634  float ax = p3.x - p2.x;
635  float ay = p3.y - p2.y;
636  float bx = p1.x - p3.x;
637  float by = p1.y - p3.y;
638  float cx = p2.x - p1.x;
639  float cy = p2.y - p1.y;
640  float apx = x - p1.x;
641  float apy = y - p1.y;

643  float bpy = y - p2.y;
644  float cpx = x - p3.x;
645  float cpy = y - p3.y;
646 
647  float aCROSSbp = ax * bpy - ay * bpx;
648  float cCROSSap = cx * apy - cy * apx;
649  float bCROSScp = bx * cpy - by * cpx;

651  return ((aCROSSbp >= 0.0f) && (bCROSScp >= 0.0f) && (cCROSSap >= 0.0f));

788  // Given a point and a triangle (defined by three points), compute the closest point
789  // in the triangle. Clamp the point so it's confined to the area of the triangle.
790  void NearestPointInTriangle(const Vector3d &point,
791  const Vector3d &triangle0,
792  const Vector3d &triangle1,
793  const Vector3d &triangle2)
794  {
795  static const Vector3d zeroVector(0, 0, 0);
796 
797  Vector3d &nearestPoint = *this;
798 
799  Vector3d lineDelta0 = triangle1 - triangle0;
800  Vector3d lineDelta1 = triangle2 - triangle0;
801 
802  // Handle degenerate triangles
803  if ((lineDelta0 == zeroVector) || (lineDelta1 == zeroVector))
804  {
805  nearestPoint.NearestPointInLineSegment(point, triangle1, triangle2);
806  }
807  else if (lineDelta0 == lineDelta1)
808  {
809  nearestPoint.NearestPointInLineSegment(point, triangle0, triangle1);
810  }
811 
812  else
813  {
814  Vector3d axis[3];
815  axis[0].NearestPointInLine(triangle0, triangle1, triangle2);
816  axis[1].NearestPointInLine(triangle1, triangle0, triangle2);
817  axis[2].NearestPointInLine(triangle2, triangle0, triangle1);
818 
819  float axisDot[3];
820  axisDot[0] = (triangle0 - axis[0]).Dot(point - axis[0]);
821  axisDot[1] = (triangle1 - axis[1]).Dot(point - axis[1]);
822  axisDot[2] = (triangle2 - axis[2]).Dot(point - axis[2]);
823 
824  bool bForce = true;
825  float bestMagnitude2 = 0;
826  float closeMagnitude2;
827  Vector3d closePoint;
828 
829  if (axisDot[0] < 0)
830  {
831  closePoint.NearestPointInLineSegment(point, triangle1, triangle2);
832  closeMagnitude2 = point.Distance2(closePoint);
833  if (bForce || (bestMagnitude2 > closeMagnitude2))
834  {
835  bForce = false;
836  bestMagnitude2 = closeMagnitude2;
837  nearestPoint = closePoint;
838  }
839  }
840  if (axisDot[1] < 0)
841  {
842  closePoint.NearestPointInLineSegment(point, triangle0, triangle2);
843  closeMagnitude2 = point.Distance2(closePoint);
844  if (bForce || (bestMagnitude2 > closeMagnitude2))
845  {
846  bForce = false;
847  bestMagnitude2 = closeMagnitude2;
848  nearestPoint = closePoint;
849  }
850  }
851  if (axisDot[2] < 0)
852  {
853  closePoint.NearestPointInLineSegment(point, triangle0, triangle1);
854  closeMagnitude2 = point.Distance2(closePoint);
855  if (bForce || (bestMagnitude2 > closeMagnitude2))
856  {
857  bForce = false;
858  bestMagnitude2 = closeMagnitude2;
859  nearestPoint = closePoint;
860  }
861  }
862 
863  // If bForce is true at this point, it means the nearest point lies
864  // inside the triangle; use the nearest-point-on-a-plane equation
865  if (bForce)
866  {
867  Vector3d normal;
868 
869  // Get the normal of the polygon (doesn't have to be a unit vector)
870  normal.Cross(lineDelta0, lineDelta1);
871 
872  Vector3d pointDelta = point - triangle0;
873  float delta = normal.Dot(pointDelta) / normal.Dot(normal);
874 
875  nearestPoint = point - normal * delta;
876  }
877  }
878  }
