485 SIMD_FORCE_INLINE btVector3 btVector3::rotate(const btVector3& wAxis, const btScalar angle) const
486 {
487  // wAxis must be a unit lenght vector
488 
489  btVector3 o = wAxis * wAxis.dot(*this);
490  btVector3 x = *this - o;
491  btVector3 y;
492 
493  y = wAxis.cross(*this);
494 
495  return (o + x * btCos(angle) + y * btSin(angle));
496 }

649 template <class T>
650 SIMD_FORCE_INLINE void btPlaneSpace1(const T& n, T& p, T& q)
651 {
652  if (btFabs(n[2]) > SIMDSQRT12)
653  {
654  // choose p in y-z plane
655  btScalar a = n[1] * n[1] + n[2] * n[2];
656  btScalar k = btRecipSqrt(a);
657  p[0] = 0;
658  p[1] = -n[2] * k;
659  p[2] = n[1] * k;
660  // set q = n x p
661  q[0] = a * k;
662  q[1] = -n[0] * p[2];
663  q[2] = n[0] * p[1];
664  }
665  else
666  {
667  // choose p in x-y plane
668  btScalar a = n[0] * n[0] + n[1] * n[1];
669  btScalar k = btRecipSqrt(a);
670  p[0] = -n[1] * k;
671  p[1] = n[0] * k;
672  p[2] = 0;
673  // set q = n x p
674  q[0] = -n[2] * p[1];
675  q[1] = n[2] * p[0];
676  q[2] = a * k;
677  }
678 }
