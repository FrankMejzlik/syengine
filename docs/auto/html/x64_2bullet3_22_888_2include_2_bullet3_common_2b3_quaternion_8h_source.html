95  void setRotation(const b3Vector3& axis, const b3Scalar& _angle)
96  {
97  b3Scalar d = axis.length();
98  b3Assert(d != b3Scalar(0.0));
99  b3Scalar s = b3Sin(_angle * b3Scalar(0.5)) / d;
100  setValue(axis.getX() * s, axis.getY() * s, axis.getZ() * s,
101  b3Cos(_angle * b3Scalar(0.5)));
102  }
107  void setEuler(const b3Scalar& yaw, const b3Scalar& pitch, const b3Scalar& roll)
108  {
109  b3Scalar halfYaw = b3Scalar(yaw) * b3Scalar(0.5);
110  b3Scalar halfPitch = b3Scalar(pitch) * b3Scalar(0.5);
111  b3Scalar halfRoll = b3Scalar(roll) * b3Scalar(0.5);
112  b3Scalar cosYaw = b3Cos(halfYaw);
113  b3Scalar sinYaw = b3Sin(halfYaw);
114  b3Scalar cosPitch = b3Cos(halfPitch);
115  b3Scalar sinPitch = b3Sin(halfPitch);
116  b3Scalar cosRoll = b3Cos(halfRoll);
117  b3Scalar sinRoll = b3Sin(halfRoll);
118  setValue(cosRoll * sinPitch * cosYaw + sinRoll * cosPitch * sinYaw,
119  cosRoll * cosPitch * sinYaw - sinRoll * sinPitch * cosYaw,
120  sinRoll * cosPitch * cosYaw - cosRoll * sinPitch * sinYaw,
121  cosRoll * cosPitch * cosYaw + sinRoll * sinPitch * sinYaw);
122  }

128  void setEulerZYX(const b3Scalar& yawZ, const b3Scalar& pitchY, const b3Scalar& rollX)
129  {
130  b3Scalar halfYaw = b3Scalar(yawZ) * b3Scalar(0.5);
131  b3Scalar halfPitch = b3Scalar(pitchY) * b3Scalar(0.5);
132  b3Scalar halfRoll = b3Scalar(rollX) * b3Scalar(0.5);
133  b3Scalar cosYaw = b3Cos(halfYaw);
134  b3Scalar sinYaw = b3Sin(halfYaw);
135  b3Scalar cosPitch = b3Cos(halfPitch);
136  b3Scalar sinPitch = b3Sin(halfPitch);
137  b3Scalar cosRoll = b3Cos(halfRoll);
138  b3Scalar sinRoll = b3Sin(halfRoll);
139  setValue(sinRoll * cosPitch * cosYaw - cosRoll * sinPitch * sinYaw, //x
140  cosRoll * sinPitch * cosYaw + sinRoll * cosPitch * sinYaw, //y
141  cosRoll * cosPitch * sinYaw - sinRoll * sinPitch * cosYaw, //z
142  cosRoll * cosPitch * cosYaw + sinRoll * sinPitch * sinYaw); //formerly yzx
143  normalize();
144  }

150  void getEulerZYX(b3Scalar& yawZ, b3Scalar& pitchY, b3Scalar& rollX) const
151  {
152  b3Scalar squ;
153  b3Scalar sqx;
154  b3Scalar sqy;
155  b3Scalar sqz;
156  b3Scalar sarg;
157  sqx = m_floats[0] * m_floats[0];
158  sqy = m_floats[1] * m_floats[1];
159  sqz = m_floats[2] * m_floats[2];
160  squ = m_floats[3] * m_floats[3];
161  rollX = b3Atan2(2 * (m_floats[1] * m_floats[2] + m_floats[3] * m_floats[0]), squ - sqx - sqy + sqz);
162  sarg = b3Scalar(-2.) * (m_floats[0] * m_floats[2] - m_floats[3] * m_floats[1]);
163  pitchY = sarg <= b3Scalar(-1.0) ? b3Scalar(-0.5) * B3_PI : (sarg >= b3Scalar(1.0) ? b3Scalar(0.5) * B3_PI : b3Asin(sarg));
164  yawZ = b3Atan2(2 * (m_floats[0] * m_floats[1] + m_floats[3] * m_floats[2]), squ + sqx - sqy - sqz);
165  }

223  b3Quaternion& operator*=(const b3Quaternion& q)
224  {
225 #if defined(B3_USE_SSE_IN_API) && defined(B3_USE_SSE)
226  __m128 vQ2 = q.get128();
227 
228  __m128 A1 = b3_pshufd_ps(mVec128, B3_SHUFFLE(0, 1, 2, 0));
229  __m128 B1 = b3_pshufd_ps(vQ2, B3_SHUFFLE(3, 3, 3, 0));
230 
231  A1 = A1 * B1;
232 
233  __m128 A2 = b3_pshufd_ps(mVec128, B3_SHUFFLE(1, 2, 0, 1));
234  __m128 B2 = b3_pshufd_ps(vQ2, B3_SHUFFLE(2, 0, 1, 1));
235 
236  A2 = A2 * B2;
237 
238  B1 = b3_pshufd_ps(mVec128, B3_SHUFFLE(2, 0, 1, 2));
239  B2 = b3_pshufd_ps(vQ2, B3_SHUFFLE(1, 2, 0, 2));
240 
241  B1 = B1 * B2; // A3 *= B3
242 
243  mVec128 = b3_splat_ps(mVec128, 3); // A0
244  mVec128 = mVec128 * vQ2; // A0 * B0
245 
246  A1 = A1 + A2; // AB12
247  mVec128 = mVec128 - B1; // AB03 = AB0 - AB3
248  A1 = _mm_xor_ps(A1, b3vPPPM); // change sign of the last element
249  mVec128 = mVec128 + A1; // AB03 + AB12
250 
251 #elif defined(B3_USE_NEON)
252 
253  float32x4_t vQ1 = mVec128;
254  float32x4_t vQ2 = q.get128();
255  float32x4_t A0, A1, B1, A2, B2, A3, B3;
256  float32x2_t vQ1zx, vQ2wx, vQ1yz, vQ2zx, vQ2yz, vQ2xz;
257 
258  {
259  float32x2x2_t tmp;
260  tmp = vtrn_f32(vget_high_f32(vQ1), vget_low_f32(vQ1)); // {z x}, {w y}
261  vQ1zx = tmp.val[0];
262 
263  tmp = vtrn_f32(vget_high_f32(vQ2), vget_low_f32(vQ2)); // {z x}, {w y}
264  vQ2zx = tmp.val[0];
265  }
266  vQ2wx = vext_f32(vget_high_f32(vQ2), vget_low_f32(vQ2), 1);
267 
268  vQ1yz = vext_f32(vget_low_f32(vQ1), vget_high_f32(vQ1), 1);
269 
270  vQ2yz = vext_f32(vget_low_f32(vQ2), vget_high_f32(vQ2), 1);
271  vQ2xz = vext_f32(vQ2zx, vQ2zx, 1);
272 
273  A1 = vcombine_f32(vget_low_f32(vQ1), vQ1zx); // X Y z x
274  B1 = vcombine_f32(vdup_lane_f32(vget_high_f32(vQ2), 1), vQ2wx); // W W W X
275 
276  A2 = vcombine_f32(vQ1yz, vget_low_f32(vQ1));
277  B2 = vcombine_f32(vQ2zx, vdup_lane_f32(vget_low_f32(vQ2), 1));
278 
279  A3 = vcombine_f32(vQ1zx, vQ1yz); // Z X Y Z
280  B3 = vcombine_f32(vQ2yz, vQ2xz); // Y Z x z
281 
282  A1 = vmulq_f32(A1, B1);
283  A2 = vmulq_f32(A2, B2);
284  A3 = vmulq_f32(A3, B3); // A3 *= B3
285  A0 = vmulq_lane_f32(vQ2, vget_high_f32(vQ1), 1); // A0 * B0
286 
287  A1 = vaddq_f32(A1, A2); // AB12 = AB1 + AB2
288  A0 = vsubq_f32(A0, A3); // AB03 = AB0 - AB3
289 
290  // change the sign of the last element
291  A1 = (b3SimdFloat4)veorq_s32((int32x4_t)A1, (int32x4_t)b3vPPPM);
292  A0 = vaddq_f32(A0, A1); // AB03 + AB12
293 
294  mVec128 = A0;
295 #else
296  setValue(
297  m_floats[3] * q.getX() + m_floats[0] * q.m_floats[3] + m_floats[1] * q.getZ() - m_floats[2] * q.getY(),
298  m_floats[3] * q.getY() + m_floats[1] * q.m_floats[3] + m_floats[2] * q.getX() - m_floats[0] * q.getZ(),
299  m_floats[3] * q.getZ() + m_floats[2] * q.m_floats[3] + m_floats[0] * q.getY() - m_floats[1] * q.getX(),
300  m_floats[3] * q.m_floats[3] - m_floats[0] * q.getX() - m_floats[1] * q.getY() - m_floats[2] * q.getZ());
301 #endif
302  return *this;
303  }
306  b3Scalar dot(const b3Quaternion& q) const
307  {
308 #if defined(B3_USE_SSE_IN_API) && defined(B3_USE_SSE)
309  __m128 vd;
310 
311  vd = _mm_mul_ps(mVec128, q.mVec128);
312 
313  __m128 t = _mm_movehl_ps(vd, vd);
314  vd = _mm_add_ps(vd, t);
315  t = _mm_shuffle_ps(vd, vd, 0x55);
316  vd = _mm_add_ss(vd, t);
317 
318  return _mm_cvtss_f32(vd);
319 #elif defined(B3_USE_NEON)
320  float32x4_t vd = vmulq_f32(mVec128, q.mVec128);
321  float32x2_t x = vpadd_f32(vget_low_f32(vd), vget_high_f32(vd));
322  x = vpadd_f32(x, x);
323  return vget_lane_f32(x, 0);
324 #else
325  return m_floats[0] * q.getX() +
326  m_floats[1] * q.getY() +
327  m_floats[2] * q.getZ() +
328  m_floats[3] * q.m_floats[3];
329 #endif
330  }

346  b3Quaternion& normalize()
347  {
348 #if defined(B3_USE_SSE_IN_API) && defined(B3_USE_SSE)
349  __m128 vd;
350 
351  vd = _mm_mul_ps(mVec128, mVec128);
352 
353  __m128 t = _mm_movehl_ps(vd, vd);
354  vd = _mm_add_ps(vd, t);
355  t = _mm_shuffle_ps(vd, vd, 0x55);
356  vd = _mm_add_ss(vd, t);
357 
358  vd = _mm_sqrt_ss(vd);
359  vd = _mm_div_ss(b3vOnes, vd);
360  vd = b3_pshufd_ps(vd, 0); // splat
361  mVec128 = _mm_mul_ps(mVec128, vd);
362 
363  return *this;
364 #else
365  return *this /= length();
366 #endif

514  b3Quaternion slerp(const b3Quaternion& q, const b3Scalar& t) const
515  {
516  b3Scalar magnitude = b3Sqrt(length2() * q.length2());
517  b3Assert(magnitude > b3Scalar(0));
518 
519  b3Scalar product = dot(q) / magnitude;
520  if (b3Fabs(product) < b3Scalar(1))
521  {
522  // Take care of long angle case see http://en.wikipedia.org/wiki/Slerp
523  const b3Scalar sign = (product < 0) ? b3Scalar(-1) : b3Scalar(1);
524 
525  const b3Scalar theta = b3Acos(sign * product);
526  const b3Scalar s1 = b3Sin(sign * t * theta);
527  const b3Scalar d = b3Scalar(1.0) / b3Sin(theta);
528  const b3Scalar s0 = b3Sin((b3Scalar(1.0) - t) * theta);
529 
530  return b3Quaternion(
531  (m_floats[0] * s0 + q.getX() * s1) * d,
532  (m_floats[1] * s0 + q.getY() * s1) * d,
533  (m_floats[2] * s0 + q.getZ() * s1) * d,
534  (m_floats[3] * s0 + q.m_floats[3] * s1) * d);
535  }
536  else
537  {
538  return *this;
539  }
540  }

552 B3_FORCE_INLINE b3Quaternion
553 operator*(const b3Quaternion& q1, const b3Quaternion& q2)
554 {
