123 // Compute the angle formed by two geodesics on the unit sphere.
124 // Three unit vectors u,v,w specify the geodesics u-v and v-w which
125 // meet at vertex uv. The angle from v-w to v-u is returned. This
126 // is always in the range [0, 2PI).
127 double SphereAngle(const VectorR3& u, const VectorR3& v, const VectorR3& w);
128 
129 // Compute the area of a triangle on the unit sphere. Three unit vectors
130 // specify the corners of the triangle in COUNTERCLOCKWISE order.
131 inline double SphericalTriangleArea(
132  const VectorR3& u, const VectorR3& v, const VectorR3& w)
133 {
134  double AngleA = SphereAngle(u, v, w);
135  double AngleB = SphereAngle(v, w, u);
136  double AngleC = SphereAngle(w, u, v);
137  return (AngleA + AngleB + AngleC - PI);
138 }

270 inline Quaternion& Quaternion::operator*=(const Quaternion& q)
271 {
272  double wnew = w * q.w - (x * q.x + y * q.y + z * q.z);
273  double xnew = w * q.x + q.w * x + (y * q.z - z * q.y);
274  double ynew = w * q.y + q.w * y + (z * q.x - x * q.z);
275  z = w * q.z + q.w * z + (x * q.y - y * q.x);
276  w = wnew;
277  x = xnew;
278  y = ynew;
279  return *this;
280 }
