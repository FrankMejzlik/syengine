200 // Resize.
201 // If the array space is decreased, the information about the allocated length is lost.
202 inline void MatrixRmn::SetSize(long numRows, long numCols)
203 {
204  assert(numRows > 0 && numCols > 0);
205  long newLength = numRows * numCols;
206  if (newLength > AllocSize)
207  {
208  delete[] x;
209  AllocSize = Max(newLength, AllocSize << 1);
210  x = new double[AllocSize];
211  }
212  NumRows = numRows;
213  NumCols = numCols;
214 }
